CREATE TABLE TransferControl
(
	TransferControlId INT IDENTITY(1,1) PRIMARY KEY,
	JobId Int,
	BatchControlNumber NVARCHAR(50),
	ReceivedDate DATETIME ,
	ProcessedDate DATETIME NULL
)

CREATE TABLE TransferControlFile
(
    TransferControlFileId INT PRIMARY KEY IDENTITY
	TransferControlId INT FOREIGN KEY REFERENCES TransferControl(TransferControlId), 
	FileLocation NVARCHAR(100) 
)
